91  void endFrame(uint16_t count)
92  {
93  // We need to send some more bytes to ensure that all the LEDs in the
94  // chain see their new color and start displaying it.
95  //
96  // The data stream seen by the last LED in the chain will be delayed by
97  // (count - 1) clock edges, because each LED before it inverts the clock
98  // line and delays the data by one clock edge. Therefore, to make sure
99  // the last LED actually receives the data we wrote, the number of extra
100  // edges we send at the end of the frame must be at least (count - 1).
101  // For the APA102C, that is sufficient.
102  //
103  // The SK9822 only updates after it sees 32 zero bits followed by one more
104  // rising edge. To avoid having the update time depend on the color of
105  // the last LED, we send a dummy 0xFF byte. (Unfortunately, this means
106  // that partial updates of the beginning of an LED strip are not possible;
107  // the LED after the last one you are trying to update will be black.)
108  // After that, to ensure that the last LED in the chain sees 32 zero bits
109  // and a rising edge, we need to send at least 65 + (count - 1) edges. It
110  // is sufficent and simpler to just send (5 + count/16) bytes of zeros.
111  //
112  // We are ignoring the specification for the end frame in the APA102/SK9822
113  // datasheets because it does not actually ensure that all the LEDs will
114  // start displaying their new colors right away.
115 
116  transfer(0xFF);
117  for (uint16_t i = 0; i < 5 + count / 16; i++)
118  {
119  transfer(0);
120  }
121  }
